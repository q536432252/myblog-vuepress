    基本语句

    # 基本语句

    # 创建表格

    drop table if exists test;
create table test (
  id int primary key,
  age int,
  name varchar(255)
);

    # 增

    insert into students (id,age,name) values (1,21,'zhangsan');

insert into students (id,age,name) values (2,22,'lisi'),(3,23,'wangwu');

insert into students values(4,24,'zhaoliu');

    # 删

    delete from students where id = 2;

    # 改

    update students set name = 'lisi' where id = 2;

    # 查

    select * from students ;

select id,age,name from students ;

    # 排序

    select * from students order by age limit 2;

select * from students order by age desc limit 2;

    # 查看第x行

    limit(第几行开始，取几行)

    select * from students order by age limit 2,1;

    # 操作表中字段：

    # #添加字段

    alter table students add email varchar(255);

    # #增加列在某列之后

    alter table user_info add school varchar(15) after level;

    # #修改字段

    #更换列的名称及数据类型(该方法要输入新的字段)
alter table user_info change oldname newname varchar(10);

    #更改数据类型(在原来字段上修改)
alter table user_info modify oldname int(11) default 0;

    # #删除字段

    alter table students drop email;

    # 修改表中的索引：

    # #普通索引

    索引名index_name可选，缺省时，MySQL将根据第一个索引列赋一个名称
#普通索引新增(普通索引要名字)
alter table visitscount add index [idx_name] (age); 
 #普通索引、唯一索引删除(普通索引要名字)
alter table visitscount drop index idx_name;

    # #唯一索引

    #唯一索引 新增
alter table visitscount add unique (age); 
#普通索引、唯一索引 删除
alter table visitscount drop index age;

    # #主键索引

    #主键索引 新增
alter table visitscount add primary key (age); 		
#主键索引 删除 （主键只有一个，不需要选择列）
alter table visitscount drop primary key;
