    # 浮点数的精度问题

    浮点数值的最高精度是17位小数，但在进行算术计算时其精确度远远不如整数。

    var result =0.1+0.2; //结果不是0.3,而是: 0.30000000000000004 
console.log(0.07 * 100); //结果不是7， 而是： 7.000000000000001

    注意：

    1. JS 中不要直接用浮点数之间进行运算，会产生精度误差。
    2. 不要直接拿两个浮点数进行比较！

    # 数组逆转

    方法名 说明 是否修改原数组
    reverse() 颠倒数组中元素顺序，无参数 会改变原数组，返回新数
    // 反转数组
function reverse(arr) {
    let res = [];
    for (let i = arr.length - 1; i >= 0; i--) {
        res[res.length] = arr[i];
    }
    return res;
}
console.log(reverse([1, 2, 3, 4, 0])); // [0, 4, 3, 2, 1]

    # 数组排序

    方法名 说明 是否修改原数组
    sort() 对数组的元素进行排序 会改变原数组，返回新数组

    sort 方法对数组进行原地排序，但是默认按照字典序排序。需要传入一个比较函数 cmp(a, b)，然后得到我们需要的排序效果。

    let arr = [1, 4, 17, 12, 9];

arr.sort();//默认按照字典序
console.log(arr); // [ 1, 12, 17, 4, 9 ]

let cmp = (a, b) => a - b;

arr.sort(cmp);
console.log(arr); // [ 1, 4, 9, 12, 17 ]Copy to clipboardErrorCopied

    其中，let cmp = (a, b) => a - b; 为升序，b - a 为降序。

    # 数组去重

    // 数组去重
function unique(arr) {
    let res = [];
    for (let i = 0; i < arr.length; i++) {
        if (res.indexOf(arr[i]) === -1) {
            res.push(arr[i]);
        }
    }
    return res;
}

console.log(unique([ 2, 3, 3, 4, 5, 5 ])); // [ 2, 3, 4, 5 ]

    # 数组转换为字符串

    方法名 说明 返回值 是否改变原数组
    toString() 将数组转换成字符串，逗号分隔每一个 返回一个字符串 不改变
    join('分隔符') 把数组中的所有元素转换为一个字符串，以指定符号分割 返回一个字符串 不改变
    let a = ['a', 'b', 'c', 'd', 'e'];

console.log(a.toString());  // a,b,c,d,e
console.log(a.join('')); // abcde
console.log(a.join('-')); // a-b-c-d-e
console.log(a); // [ 'a', 'b', 'c', 'd', 'e' ]

    # 对象操作：

    JavaScript中提供了丰富的对象操作方法，用于创建、访问、修改和删除对象的属性。以下是一些常用的对象操作方法：

    1. 创建对象：

      // 1、通过字面量创建对象
var obj = {
    name: 'zhangsan',
    age: 18,
    sayHi: function () {
        console.log('Hi!);
    }
}

// 2、使用 newObject 创建对象
// 创建空对象
let obj = new Object();
//添加属性
obj.uname = 'MurphyChen';
obj.age = 18;
obj.sayHi = function() {
    console.log('Hi!');
}
    
    

///3、通过构造函数创建对象（前两种只能一个个创建）
//构造函数名单词首字母均大写
function Star(uname, age, sex) {
    this.name = uname;
    this.age = age;
    this.sex = sex;
}
let ldh = new Star('刘德华', 18, '男');
let zxy = new Star('张学友', 19, '男');
console.log(typeof ldh); // object
console.log(ldh.sex); // 男
console.log(zxy.name); // 张学友

    2. 访问和修改属性：

      // 通过点表示法访问和修改属性
obj.name
const name = obj.name;
obj.funcName();//不要忘记括号

// 通过方括号表示法访问和修改属性
obj['name'] = value;
const name = obj['key'];

    3. 判断属性是否存在：

      // 使用in运算符判断属性是否存在
if ('key' in obj) {
  // 属性存在
}

// 使用hasOwnProperty方法判断自身属性是否存在
if (obj.hasOwnProperty('key')) {
  // 自身属性存在
}

    4. 删除属性：

      // 使用delete运算符删除属性
delete obj.key;

    5. 遍历对象属性：

      // 使用for...in循环遍历对象的可枚举属性（包括原型链上的属性）
for (let key in obj) {
    console.log(key); // 遍历属性名
    console.log(obj[key]); // 遍历属性值
}

// 使用Object.keys方法获取对象自身的可枚举属性，然后进行遍历
Object.keys(obj).forEach(function (key) {
  console.log(key, obj[key]);
});

    6. 合并对象：

      // 使用Object.assign方法将多个对象合并到目标对象
const mergedObj = Object.assign(targetObj, obj1, obj2, ...);

    7. 对象的深拷贝：

      // 使用JSON.parse和JSON.stringify进行深拷贝
const deepCopyObj = JSON.parse(JSON.stringify(obj));

    这些是JavaScript中常用的对象操作方法，可根据实际需求选择适合的方法来操作和处理对象。需要注意的是，对象是引用类型，在进行赋值和传递时需要注意引用和拷贝的问题。
